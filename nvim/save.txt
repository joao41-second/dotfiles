" Abrir o netrw em uma janela lateral automaticamente ao iniciar o Neovim
autocmd VimEnter * if !argc() | :Lexplore | endif

" Configurações do netrw
let g:netrw_banner = 0              " Desativa o banner no topo
let g:netrw_liststyle = 3           " Usa estilo de árvore para o explorador
let g:netrw_winsize = 25            " Define a largura do explorador como 25% da tela
let g:netrw_browse_split = 4        " Abre os arquivos em janelas verticais, mantendo o netrw"
let g:netrw_altv = 1                " Faz com que o explorador abra à esquerda

" Mantenha o netrw aberto quando abrir um arquivo
autocmd FileType netrw setlocal bufhidden=wipe
